#!/usr/bin/env python3 # -*- coding: utf-8 -*- import argparse import json import os from datetime import datetime import pandas as pd BURGUNDY = "#800020" MONTH_ABBR = { "1": "Jan", "2": "Feb", "3": "Mar", "4": "Apr", "5": "May", "6": "Jun", "7": "Jul", "8": "Aug", "9": "Sep", "10": "Oct", "11": "Nov", "12": "Dec" } def find_col(df, tokens_all=(), tokens_any=()): for c in df.columns: lc = str(c).lower() if all(t in lc for t in tokens_all): if not tokens_any or any(t in lc for t in tokens_any): return c return None def load_master_denominators(master_path: str): """ Load supported facilities, peer monitors, and networks from the master Excel. Uses 'Is Facility Supported' column to count only supported facilities (value = 'Yes') """ if not master_path: print("WARNING: No master file provided. Denominators will be zero.") return {"total_facilities": 0, "total_peers": 0, "total_networks": 0, "facilities": [], "peers": [], "networks": []} try: xl = pd.ExcelFile(master_path) mdf = pd.read_excel(master_path, sheet_name=xl.sheet_names[0], dtype=str).fillna("") print(f"✓ Master file loaded: {len(mdf)} rows") print(f" Columns: {list(mdf.columns)}") except Exception as e: print(f"✗ ERROR loading master file: {e}") return {"total_facilities": 0, "total_peers": 0, "total_networks": 0, "facilities": [], "peers": [], "networks": []} # Define known column names from iMonitor_Facilities.xlsx facility_col = "Facility Name" peer_col = "Peer Monitor" network_col = "Network" support_col = "Is Facility Supported" # Verify columns exist if facility_col not in mdf.columns: print(f"✗ Facility column '{facility_col}' not found!") facility_col = None if peer_col not in mdf.columns: print(f"✗ Peer column '{peer_col}' not found!") peer_col = None if network_col not in mdf.columns: print(f"✗ Network column '{network_col}' not found!") network_col = None if support_col not in mdf.columns: print(f"✗ Support column '{support_col}' not found!") support_col = None facilities, peers, networks = [], [], [] # Get supported facilities only (Is Facility Supported = Yes) if facility_col and support_col: # Filter for Yes values (case insensitive) supported_mask = mdf[support_col].astype(str).str.strip().str.upper() == "Y" supported_df = mdf[supported_mask] facilities = sorted(set(supported_df[facility_col].astype(str).str.strip() .replace("", pd.NA).dropna().tolist()), key=str.lower) print(f" Supported facilities (Yes): {len(facilities)}") elif facility_col: # Fallback to all facilities if support column not found facilities = sorted(set(mdf[facility_col].astype(str).str.strip() .replace("", pd.NA).dropna().tolist()), key=str.lower) print(f" Total facilities (no support column): {len(facilities)}") # Get all peer monitors (exclude empty) if peer_col: peers = sorted(set(mdf[peer_col].astype(str).str.strip() .replace("", pd.NA).dropna().tolist()), key=str.lower) print(f" Peer monitors: {len(peers)}") # Get all networks (exclude empty) if network_col: networks = sorted(set(mdf[network_col].astype(str).str.strip() .replace("", pd.NA).dropna().tolist()), key=str.lower) print(f" Networks: {len(networks)}") return { "total_facilities": len(facilities), "total_peers": len(peers), "total_networks": len(networks), "facilities": facilities, "peers": peers, "networks": networks, } def main(): ap = argparse.ArgumentParser( description="Build iMonitor Utilization Dashboard (HTML) — v2.9 " "(Added stacked bar and network caseload charts)" ) ap.add_argument("--input", required=True, help="Path to imonitor_enriched_*.csv") ap.add_argument("--outdir", required=True, help="Output directory") ap.add_argument("--master", default="", help="Path to iMonitor_Facilities.xlsx") args = ap.parse_args() os.makedirs(args.outdir, exist_ok=True) df = pd.read_csv(args.input, dtype=str).fillna("") # Standard columns from enrichment col_subcounty = "Subcounty" if "Subcounty" in df.columns else find_col(df, tokens_all=("subcounty",)) col_fac_raw = "Facility_raw" if "Facility_raw" in df.columns else find_col(df, tokens_all=("facility", "raw")) col_fac_master = "Facility_master" if "Facility_master" in df.columns else find_col(df, tokens_all=("facility", "master")) col_map_status = "facility_mapping_status" if "facility_mapping_status" in df.columns else find_col(df, tokens_all=("mapping", "status")) col_reported = "Reported Date" if "Reported Date" in df.columns else find_col(df, tokens_all=("reported",), tokens_any=("date",)) col_feedback = "Feedback_raw" if "Feedback_raw" in df.columns else find_col(df, tokens_all=("feedback",)) # These are guaranteed by enrichment col_reported_by = "reported_by" if "reported_by" in df.columns else None col_population = "population" if "population" in df.columns else None col_report_on = "report_on" if "report_on" in df.columns else None col_compliment_on = "compliment_on" if "compliment_on" in df.columns else None col_issue = "Issue" if "Issue" in df.columns else find_col(df, tokens_all=("issue",)) col_issue_summary = "Issue_Summary" if "Issue_Summary" in df.columns else find_col(df, tokens_all=("issue", "summary")) col_validation = "Validation Date" if "Validation Date" in df.columns else find_col(df, tokens_all=("validation",), tokens_any=("date",)) col_closure = "Case Closure Date" if "Case Closure Date" in df.columns else find_col(df, tokens_all=("closure",), tokens_any=("date",)) col_peer = "Peer Monitor" if "Peer Monitor" in df.columns else find_col(df, tokens_all=("peer",), tokens_any=("monitor", "educator", "mentor")) col_network = "Network" if "Network" in df.columns else find_col(df, tokens_all=("network",)) # Dates - FIXED: Changed dayfirst to True for dd/mm/yyyy format dt = pd.to_datetime(df[col_reported], errors="coerce", dayfirst=True) if col_reported else pd.to_datetime(pd.Series([""] * len(df)), errors="coerce", dayfirst=True) df["_dt"] = dt df["reported_date"] = df["_dt"].dt.strftime("%Y-%m-%d").fillna("") df["year"] = df["_dt"].dt.year.astype("Int64").astype(str).replace("", "") df["month_num"] = df["_dt"].dt.month.astype("Int64").astype(str).replace("", "") df["month_name"] = df["month_num"].map(lambda x: MONTH_ABBR.get(str(x), "")) df["ym"] = df["_dt"].dt.to_period("M").astype(str).replace("NaT", "") df["month_label"] = df.apply(lambda r: f"{r['month_name']} {str(r['year'])[-2:]}" if r["month_name"] and r["year"] else "", axis=1) # Facility (prefer master if mapped) if col_fac_raw: if col_fac_master and col_map_status: def choose_fac(row): if str(row.get(col_map_status, "")).strip().upper() == "MAPPED": v = str(row.get(col_fac_master, "")).strip() return v if v else str(row.get(col_fac_raw, "")).strip() return str(row.get(col_fac_raw, "")).strip() df["facility"] = df.apply(choose_fac, axis=1) else: df["facility"] = df[col_fac_raw].astype(str).str.strip() else: df["facility"] = "" def fb_type(x): s = str(x).strip().lower() if "complain" in s: return "Complaint" if "compliment" in s: return "Compliment" return "Unknown" df["feedback_type"] = df[col_feedback].map(fb_type) if col_feedback else "Unknown" df["mapping_status"] = df[col_map_status].astype(str).str.strip() if col_map_status else "" overall_min = df["_dt"].min() overall_max = df["_dt"].max() overall_min_s = overall_min.strftime("%d %b %Y") if pd.notna(overall_min) else "N/A" overall_max_s = overall_max.strftime("%d %b %Y") if pd.notna(overall_max) else "N/A" # Load denominators from master Excel master_info = load_master_denominators(args.master) total_supported_facilities = master_info["total_facilities"] total_supported_peers = master_info["total_peers"] total_supported_networks = master_info["total_networks"] records = [] for _, r in df.iterrows(): validated = (str(r.get(col_validation, "")) if col_validation else "").strip() != "" closed = (str(r.get(col_closure, "")) if col_closure else "").strip() != "" records.append({ "reported_date": str(r.get("reported_date", "")), "year": str(r.get("year", "")), "month": str(r.get("month_name", "")), "month_label": str(r.get("month_label", "")), "ym": str(r.get("ym", "")), "subcounty": str(r.get(col_subcounty, "")) if col_subcounty else "", "facility": str(r.get("facility", "")), "peer_monitor": str(r.get(col_peer, "")) if col_peer else "", "network": str(r.get(col_network, "")) if col_network else "", "mapping_status": str(r.get("mapping_status", "")), "feedback_type": str(r.get("feedback_type", "")), "reported_by": str(r.get(col_reported_by, "")) if col_reported_by else "", "report_on": str(r.get(col_report_on, "")) if col_report_on else "", "population": str(r.get(col_population, "")) if col_population else "", "compliment_on": str(r.get(col_compliment_on, "")) if col_compliment_on else "", "issue": str(r.get(col_issue, "")) if col_issue else "", "issue_summary": str(r.get(col_issue_summary, "")) if col_issue_summary else "", "validation_date": str(r.get(col_validation, "")) if col_validation else "", "closure_date": str(r.get(col_closure, "")) if col_closure else "", "validated": "Yes" if validated else "No", "closed": "Yes" if closed else "No", "case_status": "Closed" if closed else "Pending closure" }) ts = datetime.now().strftime("%Y%m%d_%H%M%S") out_html = os.path.join(args.outdir, f"iMonitor_Utilization_Dashboard_GF_TB_v2_9_{ts}.html") #
